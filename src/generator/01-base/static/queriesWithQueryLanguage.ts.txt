const flattenOrderBy = (orderBy: OrderBy<any>[] = []): string => {
  const orderByModifierKeys: (keyof OrderByModifier)[] = ['SORT', 'LOWER', 'TRIM', 'LENGTH'];

  const applyModifiers = (property: string, modifier: { LENGTH?: boolean; TRIM?: boolean; LOWER?: boolean }) => {
    let result = property;
    if (modifier.TRIM) {
      result = `trim(${result})`;
    }
    if (modifier.LOWER) {
      result = `lower(${result})`;
    }
    if (modifier.LENGTH) {
      result = `length(${result})`;
    }
    return result;
  };

  const isConditionalOrderBy = (
    orderByItem: OrderBy<any> | ConditionalOrderBy<any, boolean> | undefined
  ): orderByItem is ConditionalOrderBy<any, boolean> =>
    !!orderByItem && typeof orderByItem === 'object' && 'CASE' in orderByItem;

  const isConditionalCaseValue = (
    value: QueryFilter<any> | ConditionalOrderBy<any, boolean>
  ): value is ConditionalOrderBy<any, boolean> =>
    !!value && typeof value === 'object' && 'CASE' in value;

  const applyModifiersToConditionProperties = (
    condition: string,
    modifier?: OrderByModifier
  ): string => {
    if (!modifier || (!modifier.LOWER && !modifier.TRIM && !modifier.LENGTH)) {
      return condition;
    }

    return condition.replace(
      /([A-Za-z_][A-Za-z0-9_.]*)\s*(=|!=|<=|>=|<|>|~|in|null)\b/g,
      (_fullMatch, property: string, operator: string) =>
        `${applyModifiers(property, modifier)} ${operator}`
    );
  };

  const flattenSingle = (
    orderByItem: OrderBy<any> | ConditionalOrderBy<any, boolean> | undefined,
    isRootConditional = false
  ): string[] => {
    if (isConditionalOrderBy(orderByItem)) {
      const [, when, , then, , elseVal, modifier] = orderByItem.CASE;
      const whenStr = isConditionalCaseValue(when)
        ? flattenSingle(when, false)[0]
        : applyModifiersToConditionProperties(
          flattenWhere(when as QueryFilter<any>, []).join(' and '),
          modifier
        );
      const thenStr = typeof then === 'number'
        ? then.toString()
        : flattenSingle(then, false)[0];
      const elseStr = typeof elseVal === 'number'
        ? elseVal.toString()
        : flattenSingle(elseVal, false)[0];
      let expr = `(${whenStr}) ? ${thenStr} : ${elseStr}`;
      if (isRootConditional) {
        expr += modifier?.SORT === 'desc' ? ' desc' : ' asc';
      }
      return [expr];
    }
    const result: string[] = [];
    for (const key in orderByItem) {
      const value = (orderByItem as any)[key];
      if (!value) continue;

      if (typeof value === 'object' && !Array.isArray(value)) {
        const modifierValue = value as OrderByModifier;
        const hasModifier = orderByModifierKeys.some((modifierKey) => modifierKey in modifierValue);

        if (hasModifier) {
          let propExpr = key;
          propExpr = applyModifiers(propExpr, modifierValue);
          result.push(propExpr + (modifierValue.SORT === 'desc' ? ' desc' : ' asc'));
          continue;
        }

        const nestedValue = (value as any)[key];
        if (nestedValue && typeof nestedValue === 'object' && !Array.isArray(nestedValue)) {
          let propExpr = key;
          propExpr = applyModifiers(propExpr, nestedValue);
          result.push(propExpr + (nestedValue.SORT === 'desc' ? ' desc' : ' asc'));
        }
      }
    }
    return result;
  };

  return orderBy.flatMap((item) => flattenSingle(item, true)).join(', ');
}

export type ComparisonOperator =
  | 'EQ'
  | 'NE'
  | 'LT'
  | 'GT'
  | 'LE'
  | 'GE'
  | 'LIKE';

export type LengthOperator = 'LENGTH';

export type ArrayOperator = 'IN';

export type NullOperator = 'NULL';

export type Operator = ComparisonOperator | ArrayOperator | NullOperator;

export type ModifierFunction = 'LOWER' | 'TRIM';

// use globalThis to use typescript's 'Pick', since the SDK contains an interface which is also called 'Pick'
export type RequireAtLeastOne<T> = {
  [K in keyof T]-?: Required<globalThis.Pick<T, K>> & Partial<Omit<T, K>>;
}[keyof T];

export type LengthExpr = {
  [K in LengthOperator]: {
    [K in Exclude<ComparisonOperator, 'LIKE'>]?: number
  }
};

export type LengthExprWithModifier =
  LengthExpr &
  { [K in ComparisonOperator]?: never } &
  { [K in ArrayOperator]?: never } &
  { [K in NullOperator]?: never } &
  { [K in ModifierFunction]?: boolean };

export type FilterExpr<T> =
  { [K in ComparisonOperator]?: T } &
  { [K in ArrayOperator]?: T[] } &
  { [K in keyof LengthExpr]?: never };

export type FilterExprWithoutNull<T> =
  RequireAtLeastOne<FilterExpr<T>> &
  { [K in NullOperator]?: never } &
  { [K in ModifierFunction]?: boolean };

export type FilterExprWithNull<T> =
  FilterExpr<T> &
  { [K in NullOperator]?: boolean} &
  { [K in ModifierFunction]?: never };

export type MapOperators<T> = LengthExprWithModifier | FilterExprWithoutNull<T> | FilterExprWithNull<T>;

export type SingleFilterExpr<T> = {
  [P in keyof T]?: T[P] extends Array<infer U> | undefined
    ? U extends Record<any, any>
      ? SingleFilterExpr<U> | { NOT?: SingleFilterExpr<U> }
      : MapOperators<U>
    : T[P] extends Record<any, any> | undefined
      ? SingleFilterExpr<T[P]> | { NOT?: SingleFilterExpr<T[P]> }
      : MapOperators<T[P]>;
};

export type QueryFilter<T> = SingleFilterExpr<T> & {
  OR?: QueryFilter<T>[];
  AND?: QueryFilter<T>[];
  NOT?: QueryFilter<T>;
};

export type OrderByModifier = {
  SORT: 'asc' | 'desc';
  LOWER?: boolean;
  TRIM?: boolean;
  LENGTH?: boolean;
};

type RootConditionalOrderByModifier = Omit<OrderByModifier, 'SORT'> & {
  SORT?: 'asc' | 'desc';
};

type NestedConditionalOrderByModifier = Omit<OrderByModifier, 'SORT'>;

export type ConditionalOrderBy<E, IsRoot extends boolean = true> = {
  CASE: [
    'WHEN', QueryFilter<E> | ConditionalOrderBy<E, false>,
    'THEN', number | ConditionalOrderBy<E, false>,
    'ELSE', number | ConditionalOrderBy<E, false>,
    (IsRoot extends true
      ? RootConditionalOrderByModifier
      : NestedConditionalOrderByModifier)?
  ];
};

export type PropertyOrderBy<E> = {
  [K in keyof E]?: {
    [V in keyof E]?: V extends K
      ? E[V] extends Array<infer U> | undefined
        ? U extends object
          ? Sort<U>
          : never
        : E[V] extends object | undefined
          ? Sort<E[V]>
          : OrderByModifier
      : never;
  };
}[keyof E];

export type OrderBy<E> = PropertyOrderBy<E> | ConditionalOrderBy<E, true>;

export type CountQuery<F> = {
  where?: QueryFilter<F>;
};

export type SomeQuery<E, F, I, P> = {
  serializeNulls?: boolean;
  include?: QuerySelect<I>;
  properties?: P;
  where?: QueryFilter<F>;
  select?: QuerySelect<E>;
  sort?: Sort<E>[];
  pagination?: Pagination;
  orderBy?: OrderBy<E>[];
};

const comparisonOperatorList: ComparisonOperator[] = [
  'EQ',
  'NE',
  'LT',
  'GT',
  'LE',
  'GE',
  'LIKE'
];

const comparisonOperatorMap: Record<Operator, string> = {
  EQ: '=',
  NE: '!=',
  LT: '<',
  GT: '>',
  LE: '<=',
  GE: '>=',
  LIKE: '~',
  IN: 'in',
  NULL: 'null'
};

const modifierFunctionList: ModifierFunction[] = ['LOWER', 'TRIM'];

const flattenWhere = (
  obj: QueryFilter<any> = {},
  nestedPaths: string[]
): string[] => {
  const entries: string[] = [];
  for (const [prop, propValue] of Object.entries(obj)) {
    const setModifiers = findAllModifierFunctions(propValue ?? {}, modifierFunctionList).filter(
      (modifier) => modifier[1]
    );
    if (prop === 'OR') {
      const flattedOr: string[][] = [];
      for (let i = 0; i < (obj.OR?.length ?? 0); i++) {
        flattedOr.push(flattenWhere(obj.OR?.[i], nestedPaths));
      }
      entries.push(
        `(${flattedOr
          .map((x) => {
            const joined = x.join(' and ');

            if (x.length > 1) {
              return `(${joined})`;
            } else {
              return joined;
            }
          })
          .join(' or ')})`
      );
    } else if (prop === 'AND') {
      const flattedAnd: string[][] = [];
      for (let i = 0; i < (obj.AND?.length ?? 0); i++) {
        flattedAnd.push(flattenWhere(obj.AND?.[i], nestedPaths));
      }
      entries.push(
        `(${flattedAnd
          .map((x) => {
            const joined = x.join(' and ');

            if (x.length > 1) {
              return `(${joined})`;
            } else {
              return joined;
            }
          })
          .join(' and ')})`
      );
    } else if (prop === 'NOT') {
      const flattedNot = flattenWhere(obj.NOT, nestedPaths);
      entries.push(
        `not ${flattedNot.length > 1 ? '(' : ''}${flattedNot.join(' and ')}${flattedNot.length > 1 ? ')' : ''}`
      );
    } else if (propValue) {
      for (const [operator, value] of Object.entries(propValue)) {
        if (value === undefined) continue;
        if (comparisonOperatorList.includes(operator as ComparisonOperator)) {
          entries.push(
            `${setModifiers.reduce(
              (acc, [first]) => `${first.toLowerCase()}(${acc})`,
              nestedPaths.some((path) => path === prop) ? nestedPaths.join('.') : [...nestedPaths, prop].join('.')
            )} ${comparisonOperatorMap[operator as Operator]} ${
              typeof value === 'string' ? JSON.stringify(value) : value
            }`
          );
        } else if ((operator as Operator) === 'NULL') {
          entries.push(
            `${!value ? 'not ' : ''}${nestedPaths.some((path) => path === prop) ? nestedPaths.join('.') : [...nestedPaths, prop].join('.')} ${comparisonOperatorMap[operator as Operator]}`
          );
        } else if ((operator as Operator) === 'IN') {
          if(value.length === 0) {
            entries.push('1 = 0')
          } else {
            entries.push(
              `${setModifiers.reduce(
                (acc, [first]) => `${first.toLowerCase()}(${acc})`,
                nestedPaths.some((path) => path === prop) ? nestedPaths.join('.') : [...nestedPaths, prop].join('.')
              )} ${comparisonOperatorMap[operator as Operator]} [${value.map((v: string | number) =>
                typeof v === 'string' ? JSON.stringify(v) : v
              )}]`
            );
          }
        } else if ((operator as LengthOperator) === 'LENGTH') {
          const lengthProp = `length(${setModifiers.reduce(
            (acc, [first]) => `${first.toLowerCase()}(${acc})`,
            nestedPaths.some((path) => path === prop) ? nestedPaths.join('.') : [...nestedPaths, prop].join('.')
          )})`;
          entries.push(...flattenWhere({ [lengthProp]: value } as QueryFilter<any>, []));
        } else if (
          !modifierFunctionList.includes(operator as ModifierFunction)
          && typeof value === 'object'
        ) {
          entries.push(
            ...flattenWhere(propValue as QueryFilter<any>, [
              ...nestedPaths,
              prop
            ])
          );
          break;
        }
      }
    }
  }
  return entries;
};

const assembleOrderBy = (orderBy: OrderBy<any>[] = []): Record<string, string> {
  if(!orderBy.length) {
    return {}
  }
  const flattedOrderBy = flattenOrderBy(orderBy);
  return flattedOrderBy.length ? { orderBy: flattedOrderBy } : {};
}

const assembleFilterParam = (
  obj: QueryFilter<any> = {}
): Record<string, string> => {
  const flattedFilter = flattenWhere(obj, []);
  return flattedFilter.length ? { filter: flattedFilter.join(' and ') } : {};
};

const findAllModifierFunctions = (
  obj: Record<string, any>,
  types: ModifierFunction[]
) => {
  const result: [string, any][] = [];
  for (const key in obj) {
    if (types.includes(key as ModifierFunction)) {
      result[types.indexOf(key as ModifierFunction)] = [key, obj[key]];
    }
  }
  return result.filter((modifierTuple) => modifierTuple);
};

const _count = (
  cfg: ServiceConfig | undefined,
  endpoint: string,
  query?: CountQuery<any> & { params?: Record<any, any> },
  requestOptions?: RequestOptions
) =>
  {
    const usePost = cfg?.usePost ?? globalConfig?.usePost
    const payload = {
      ...assembleFilterParam(query?.where),
      ...query?.params
    }

    return wrapResponse(() =>
      raw(cfg, endpoint, {
        unwrap: true,
        method: usePost ? 'POST' : 'GET',
        ...(usePost ? { body: payload } : { query: payload })
      }, requestOptions)
    )
  };

const _some = (
  cfg: ServiceConfig | undefined,
  endpoint: string,
  query?: SomeQuery<any, any, any, any> & { params?: Record<any, any> },
  requestOptions?: RequestOptions
) => {
  const usePost = cfg?.usePost ?? globalConfig?.usePost;
  const payload = {
    serializeNulls: query?.serializeNulls,
    additionalProperties: query?.properties?.join(','),
    properties: query?.select
      ? flattenSelect(query.select).join(',')
      : undefined,
    includeReferencedEntities: query?.include
      ? Object.keys(query.include).join(',')
      : undefined,
    ...assembleFilterParam(query?.where),
    ...flattenSort(query?.sort),
    ...assembleOrderBy(query.orderBy),
    ...query?.params,
    ...query?.pagination
  };
  return wrapResponse(() =>
    raw(cfg, usePost ? `${endpoint}/query` : endpoint, {
      method: usePost ? 'POST' : 'GET',
      ...(usePost ? { body: payload } : { query: payload })
    }, requestOptions).then((data) => ({
      entities: data.result,
      references: data.referencedEntities ?? {},
      properties: data.additionalProperties ?? {}
    }))
  );
};
